    ? `<ul class="name-list">${arr.map((n,i) => `<li><span class="dot ${cls}"></span>${i+1}. ${n}</li>`).join('')}</ul>`

    : `<div class="none-msg">None</div>`;

  grid.innerHTML += `

    <div class="office-card">

      <div class="office-hdr ${d.remote ? 'remote' : ''}">

        <div class="office-hdr-name">${d.remote ? '🌐' : '🏙️'} ${d.office}</div>

        <span class="office-badge">${d.yes.length} attending</span>

      </div>

      <div class="office-body">

        <div class="sub-hdr yes">✅ Attending — ${d.yes.length}</div>

        ${mkList(d.yes, 'yes')}

        <div class="sub-hdr no">❌ Not Attending — ${d.no.length}</div>

        ${mkList(d.no, 'no')}

        <div class="sub-hdr nonr">❓ No Response — ${d.nonr.length}</div>

        ${mkList(d.nonr, 'nonr')}

      </div>

    </div>`;

});
